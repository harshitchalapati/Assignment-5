def minimax(game, maximizing):

    result = game.winner()

    if result == 'X':
        return 1
    elif result == 'O':
        return -1
    elif result == 'Draw':
        return 0

    if maximizing:
        best = -float('inf')

        for move in game.available_moves():
            temp = game.clone()
            temp.make_move(move, 'X')

            score = minimax(temp, False)

            best = max(best, score)

        return best

    else:
        best = float('inf')

        for move in game.available_moves():
            temp = game.clone()
            temp.make_move(move, 'O')

            score = minimax(temp, True)

            best = min(best, score)

        return best
